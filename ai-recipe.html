async function generateRecipe() { document.getElementById("result").innerText = "Testing..."; try { let res = await fetch("/api/recipe", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ ingredients: "test" }) }); let data = await res.json(); document.getElementById("result").innerText = JSON.stringify(data, null, 2); } catch (err) { document.getElementById("result").innerText = "FAILED: " + err.message; } }
